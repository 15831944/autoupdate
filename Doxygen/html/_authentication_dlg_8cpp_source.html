201 CString CAuthenticationDlg::GetActiveUser()
202 {
203  // Declare variables
204  CString sUsername = _T(""), sDomainName = _T(""), sBuffer, sResult = _T("");
205  PSID pSid = NULL;
206  BYTE bySidBuffer[256];
207  DWORD dwSidSize = sizeof(bySidBuffer), dwDomainNameSize = 1024, dwUsernameSize = 1024;
208  SID_NAME_USE sidType;
209 
210  // Initialize variables
211  pSid = (PSID)bySidBuffer;
212  dwSidSize = sizeof(bySidBuffer);
213 
214  // Get current username
215  GetUserName(sBuffer.GetBuffer(dwUsernameSize), &dwUsernameSize);
216  sUsername = sBuffer.GetBuffer(dwUsernameSize);
217  sBuffer.ReleaseBuffer();
218 
219  // Get the domain name
220  LookupAccountName(NULL, sUsername, (PSID) pSid, &dwSidSize,
221  sBuffer.GetBuffer(dwDomainNameSize), &dwDomainNameSize, (PSID_NAME_USE)&sidType);
222  sDomainName = sBuffer.GetBuffer(dwDomainNameSize);
223  sBuffer.ReleaseBuffer();
224 
225  // Return right result
226  if (!sDomainName.IsEmpty())
227  sResult = sDomainName + _T("\\");
228  sResult += sUsername;
229 
230  return sResult;
231 }
