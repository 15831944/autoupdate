65 /* MD5_F, MD5_G and MD5_H are basic MD5 functions: selection, majority, parity */
66 #define MD5_F(x, y, z) (((x) & (y)) | ((~x) & (z)))
67 #define MD5_G(x, y, z) (((x) & (z)) | ((y) & (~z)))
68 #define MD5_H(x, y, z) ((x) ^ (y) ^ (z))
69 #define MD5_I(x, y, z) ((y) ^ ((x) | (~z)))

76 /* MD5_FF, MD5_GG, MD5_HH, and MD5_II transformations for rounds 1, 2, 3, and 4 */
77 /* Rotation is separate from addition to prevent recomputation */
78 #define MD5_FF(a, b, c, d, x, s, ac) {(a) += MD5_F ((b), (c), (d)) + (x) + (UINT4)(ac); (a) = ROTATE_LEFT ((a), (s)); (a) += (b); }

196 void MD5Update (MD5_CTX *mdContext, unsigned char *inBuf, unsigned int inLen)
197 {
198  UINT4 in[16];
199  int mdi = 0;
200  unsigned int i = 0, ii = 0;
201 
202  /* Compute number of bytes mod 64 */
203  mdi = (int)((mdContext->i[0] >> 3) & 0x3F);
204 
205  /* Update number of bits */
206  if ((mdContext->i[0] + ((UINT4)inLen << 3)) < mdContext->i[0])
207  mdContext->i[1]++;
208  mdContext->i[0] += ((UINT4)inLen << 3);
209  mdContext->i[1] += ((UINT4)inLen >> 29);
210 
211  while (inLen--)
212  {
213  /* Add new character to buffer, increment mdi */
214  mdContext->in[mdi++] = *inBuf++;
215 
216  /* Transform if necessary */
217  if (mdi == 0x40)
218  {
219  for (i = 0, ii = 0; i < 16; i++, ii += 4)
220  in[i] = (((UINT4)mdContext->in[ii+3]) << 24) |
221  (((UINT4)mdContext->in[ii+2]) << 16) |
222  (((UINT4)mdContext->in[ii+1]) << 8) |
223  ((UINT4)mdContext->in[ii]);
224 
225  MD5_Transform (mdContext->buf, in);
226  mdi = 0;

231 void MD5Final (MD5_CTX *mdContext)
232 {
233  UINT4 in[16];
234  int mdi = 0;
235  unsigned int i = 0, ii = 0, padLen = 0;
236 
237  /* Save number of bits */
238  in[14] = mdContext->i[0];
239  in[15] = mdContext->i[1];
240 
241  /* Compute number of bytes mod 64 */
242  mdi = (int)((mdContext->i[0] >> 3) & 0x3F);
243 
244  /* Pad out to 56 mod 64 */
245  padLen = (mdi < 56) ? (56 - mdi) : (120 - mdi);
246  MD5Update (mdContext, MD5_PADDING, padLen);
247 
248  /* Append length in bits and transform */
249  for (i = 0, ii = 0; i < 14; i++, ii += 4)
250  in[i] = (((UINT4)mdContext->in[ii+3]) << 24) |
251  (((UINT4)mdContext->in[ii+2]) << 16) |
252  (((UINT4)mdContext->in[ii+1]) << 8) |
253  ((UINT4)mdContext->in[ii]);
254  MD5_Transform (mdContext->buf, in);
255 
256  /* Store buffer in digest */
257  for (i = 0, ii = 0; i < 4; i++, ii += 4)
258  {
259  mdContext->digest[ii] = (unsigned char)( mdContext->buf[i] & 0xFF);
260  mdContext->digest[ii+1] = (unsigned char)((mdContext->buf[i] >> 8) & 0xFF);
261  mdContext->digest[ii+2] = (unsigned char)((mdContext->buf[i] >> 16) & 0xFF);
262  mdContext->digest[ii+3] = (unsigned char)((mdContext->buf[i] >> 24) & 0xFF);
263  }
264 }
