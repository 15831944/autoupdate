114 void CInformationDlg::AutoResizeControls()
115 {
116  // Declare variables
117  CString sWindowText;
118  CRect rectLabel, rectOK, rectDialog;
119  int iButtonHeight;
120  CDC * pDC = m_lblExplanation.GetDC();
121 
122  // Get window text & size
123  m_lblExplanation.GetWindowText(sWindowText);
124  m_lblExplanation.GetWindowRect(&rectLabel);
125 
126  // Get size of explanation
127  pDC->SelectObject(GetFont());
128  pDC->DrawText(sWindowText, &rectLabel, DT_CALCRECT | DT_WORDBREAK);
129 
130  // Clear memory
131  //delete pDC; <-- causes crash at end of application
132 
133  // Resize our label
134  ScreenToClient(&rectLabel);
135  m_lblExplanation.MoveWindow(&rectLabel);
136 
137  // Now get the positions of our controls
138  ClientToScreen(&rectLabel);
139  m_btnOK.GetWindowRect(&rectOK);
140  GetWindowRect(&rectDialog);
141 
142  // Get height of buttons
143  iButtonHeight = rectOK.bottom - rectOK.top;
144 
145  // Calculate new positions
146  rectOK.top = rectLabel.bottom + 13;
147  rectOK.bottom = rectOK.top + iButtonHeight;
148  rectDialog.bottom = rectOK.bottom + 13;
149 
150  // Set new position of buttons
151  ScreenToClient(&rectOK);
152  m_btnOK.MoveWindow(&rectOK);
153 
154  // Set new position of dialog
155  MoveWindow(&rectDialog);
156 }
