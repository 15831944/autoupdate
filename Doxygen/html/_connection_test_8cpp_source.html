83 bool CConnectionTest::AutoDialSet(const BOOL bOn)
84 {
85  // Declare variables
86  HKEY hKey;
87  DWORD dwDial = -1;
88  bool bStat = false;
89 
90  if (RegOpenKeyEx(HKEY_CURRENT_USER, gszInternetSettingsSubKey,
91  0, KEY_SET_VALUE | KEY_QUERY_VALUE, &hKey) == ERROR_SUCCESS)
92  {
93  const int iDialSize = 4;
94  DWORD dwDialType = REG_BINARY, dwDialSize = iDialSize;
95  BYTE byDial[iDialSize];
96 
97  // We only want to change byte 0 so get the current 4 bytes
98  if (RegQueryValueEx(hKey, gszEnableAutodialItem, NULL, &dwDialType, (LPBYTE)&byDial, &dwDialSize) == ERROR_SUCCESS)
99  {
100  // Users may already have dodgy Reg Entries so don't do this in public
101  // ASSERT( dwDialType == REG_BINARY && dwDialSize == 4 ); // Let us know if MS ever change how/what's stored
102 
103  // REG_BINARY is what windows uses, however some apps can
104  // incorrectly change it to REG_DWORD.
105  if ((dwDialType != REG_BINARY) || (dwDialSize != 4))
106  {
107  // If it has been changed to REG_DWORD we may as well leave it that way as our code works eitehr way.
108  if (dwDialType == REG_DWORD)
109  ZeroMemory(byDial, iDialSize);
110  }
111 
112  // This works correctly for both REG_BINARY and REG_DWORD. For REG_DWORD we get 0x00000001
113  byDial[0] = bOn ? 1 : 0; // turn it on or off
114 
115  if (RegSetValueEx(hKey, gszEnableAutodialItem, NULL, dwDialType, (LPBYTE)&byDial, dwDialSize) == ERROR_SUCCESS)
116  bStat = true;
117  }
118  RegCloseKey(hKey);
119  }
120 
121  return bStat;
122 }

143 
144  // See comments in AutoDialSet()

146  nDial = byDial[0] == 0 ? 0 : 1;
147  else

150  nDial = byDial[0] == 0 ? 0 : 1; // for REG_DWORD Low Byte is also what we want.
151  }

154  }

161 bool CConnectionTest::TestConnection(CString sURL)
162 {
163  USES_CONVERSION;
164 
165  // Declare variables
166  LPHOSTENT hostEntry;
167  SOCKADDR_IN serverInfo;
168  int iResult;
169  SOCKET testSocket;
170 
171  // Get information about server
172  hostEntry = gethostbyname(W2A(sURL));
173 
174  // Check if server information is retrieved successfully
175  if (!hostEntry)
176  return false;
177 
178  // Setup socket
179  testSocket = socket(AF_INET, // Use TCP/IP
180  SOCK_STREAM, // Stream-oriented
181  IPPROTO_TCP); // Use TCP protocol
182 
183 
184  // Check if socket is created
185  if (testSocket == INVALID_SOCKET)
186  {
187  // Close socket
188  closesocket(testSocket);
189  return false;
190  }
191 
192  // Fill in address information
193  serverInfo.sin_family = AF_INET;
194  serverInfo.sin_addr = *((LPIN_ADDR)*hostEntry->h_addr_list);
195  serverInfo.sin_port = htons(INTERNET_DEFAULT_HTTP_PORT);
196 
197  // Now connect to server
198  iResult = connect(testSocket, (LPSOCKADDR)&serverInfo, sizeof(struct sockaddr));
199 
200  // Close socket
201  closesocket(testSocket);
202 
203  // Check result
204  if (iResult == SOCKET_ERROR)
205  return false;
206 
207  // We did it!
208  return true;
209 }
