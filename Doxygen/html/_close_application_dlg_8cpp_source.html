173 void CCloseApplicationDlg::OnTimer(UINT nIDEvent)
174 {
175  // Declare variables
176  CString sTemp;
177  bool bRunning = false;
178 
179  // Did we receive the right timer event?
180  if (nIDEvent == TIMER_COUNTDOWN)
181  {
182  // Check if the application is still running
183  std::string strVersion;
184  UINT uWinPlatform;
185  UINT uVersion = getWinVersion(strVersion, uWinPlatform);
186  CString sProcess = CPath::Instance()->ExtractFileName(m_sApplicationPath);
187  if ((uVersion == WINDOWS_NT3) || (uVersion == WINDOWS_NT4))
188  {
189  bRunning = NTApplicationRunning(sProcess.GetBuffer(sProcess.GetLength()));
190  }
191  else
192  {
193  bRunning = AllWinApplicationRunning(sProcess.GetBuffer(sProcess.GetLength()));
194  }
195 
196  // Is the application running?
197  if (bRunning == false)
198  {
199  // No, we can close the dialog
200  EndDialog(IDOK);
201  }
202 
203  // Lower the timer
204  m_iCountDown--;
205 
206  // Set up text
207  if (m_iCountDown > 0)
208  {
209  // Set explanation text
210  sTemp.Format(CLanguage::Instance()->GetString(IDS_CLOSEAPPLICATION_EXPLANATION), m_sApplicationTitle,
211  m_iCountDown, m_sApplicationTitle);
212  m_lblExplanation.SetWindowText(sTemp);
213 
214  // Update progressbar
215  m_prgEnd.SetPos(m_prgEnd.GetPos() + 1);
216  }
217  else
218  {
219  // Kill timer
220  KillTimer(TIMER_COUNTDOWN);
221 
222  // Force application to close
223  OnBtnEndNow();
224  }
225  }
226 }
